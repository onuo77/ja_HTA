create table users(
	user_id			varchar2(20) primary key, -- not null unique
	user_name		varchar2(20) not null,
	user_password		varchar2(20) not null,	-- null을 허용하지 않음
	user_photo_filename 	varchar2(100),		-- null을 허용함
	user_email		varchar2(100) not null unique,
	user_phone		varchar2(20) unique,
	user_postal_code	char(5),
	user_address		varchar2(200),
	user_age		number(3) check (user_age >= 1 and user_age <= 200),
	user_gender		char(1) check (user_gender in ('F', 'M'))
	user_grade		varchar(20) check (user_grade in ('일반', '로얄', '골드', '플래티넘'))
)

not null
	해당 컬럼의 값을 null을 허용하지 않는다.
	테이블에 여러번 지정할 수 있다.

unique (고유키 제약조건)
	해당 컬럼의 값은 테이블전체에서 고유한 값이어야 한다.
	null이 여러 행에 중복 입력되는 것을 허용한다.
	테이블에 여러번 지정할 수 있다.

primary key(기본키 제약조건)
	해당 컬럼의 값은 null이 허용되지 않으며, 테이블 전체에서 고유한 값이어야 한다.
	테이블 당 하나만 지정할 수 있다.
	테이블에서 행을 대표하는 컬럼에 지정한다.

check 제약조건
	해당 컬럼의 값은 제약조건에서 지정한 조건식을 만족하는 값만 허용된다.
	테이블에 여러번 지정할 수 있다.

foreign key 제약조건 (외래키 제약조건, 참조키 제약조건)
	해당 컬럼의 값은 다른 테이블 혹은 같은 테이블의 특정 컬럼이 가지고 있는 값만 허용된다.
	종속 테이블(자식 테이블) : FOREIGN KEY로 설정된 컬럼은 부모테이블의 기존 값과 일치하거나 NULL이어야 한다.
	참조 테이블(부모 테이블) : 자식테이블이 참조하는 부모테이블의 컬럼은 PRIMARY KEY 혹은 UNIQUE 제약조건이 정의된 컬럼이어야 한다.
